// TO REVOKE THE FREEZE AUTHORITY:
spl-token authorize <TOKEN_MINT_ADDRESS> freeze --disable   // freeze authority set to null
// TO CHANGE THE FREEZE AUTHORITY:
spl-token authorize <TOKEN_MINT_ADDRESS> freeze <NEW_FREEZE_AUTHORITY_ADDRESS>
